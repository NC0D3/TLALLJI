def escribir(nombre_archivo, *lineas):
    # Abre el archivo en modo escritura ('w')
    with open(nombre_archivo, 'w') as archivo:
        # Escribe cada línea en el archivo
        for linea in lineas:
            archivo.write(linea + '\n')  # Agrega un salto de línea después de cada línea

# Ejemplo de uso
escribir('mi_archivo.txt', 'Hola', 'Esta es otra línea', 'Tercera línea')
